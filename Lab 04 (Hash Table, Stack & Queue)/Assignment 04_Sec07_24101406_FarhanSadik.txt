            String x ="N";
            int temp = (int)x.charAt(0);
            y += Integer.toString(temp);
            sum += Integer.parseInt(y);
        }

    public String searchHashtable( Object[] keyValuePair ){
        String x = (String)keyValuePair[0];
        int indexOfHash = hashFunction(x);
        PairNode current = ht[indexOfHash];
        while (current != null){
            if(current.key == x) return "Found";
            current = current.next;
        }

    private int hashFunction( String key ){
        int sum=0;
        for(int i=0; i<key.length(); i++){
            if(key.length()%2==0 && i%2==0){
                char ch = key.charAt(i);
                sum += (int)ch;
            }
            else if(key.length()%2!=0 && i%2!=0){
                char ch = key.charAt(i);
                sum += (int)ch;
            }
        }
        return sum%ht.length;
    }

    public void insert(String key, Integer value){
        int index = hashFunction(key);
        FruitNode newOne = new FruitNode(key, value);
        if(ht[index] == null){
            ht[index] = newOne;
        }
        else{
            FruitNode current = ht[index];
            boolean check =true;
            while (current.next!=null) {
                if(current.fruit[0] == key){
                    if((int)current.fruit[1] >= value){
                        check = false;
                        break;
                    }
                }
                current = current.next;
            }
            if(check){
                current.next = newOne;
            }
        }
    }

    private int hashFunction( Integer key ){
        return (key + 3) % 6;
    }

    public void remove( Integer key ){
        int index = hashFunction(key);
        if(ht[index] == null){
            return;
        }
        PairNode current = ht[index];
        PairNode temp = null;
        while (current!= null) {
            if(current.key == key && temp == null){
                ht[index] = current.next;
                break;
            }
            if(current.key == key){
                temp.next = current.next;
            }
            temp = current;
            current = current.next;
        }        
    }

    public static int diamondCount(Stack stack, String str) {
        int count =0;
        for(int i=0; i<str.length(); i++){
            char ch = str.charAt(i);
            if(ch=='<'){
                stack.push(ch);
            }
            else if(ch=='>'){
                if(!stack.isEmpty()){
                    stack.pop();
                    count++;
                }
            }
        }
        return count;
    }





//Task 5
    public static void removeBlock(Stack stack, int n) {
        Stack temp = new Stack();
        for(int i=0; i<n-1; i++){
            temp.push(stack.pop());
        }
        stack.pop();
        for(int i=0; i<n-1; i++){
            stack.push(temp.pop());
        }
    }





//Task 6
    public static Stack conditionalReverse(Stack stack) {
        Stack temp = new Stack();
        while (!stack.isEmpty()){
            int x = stack.pop();
            if(!temp.isEmpty() && x==temp.peek()){
                continue;
            }
            else{
                temp.push(x);
            }
        }
        return temp;
    }
